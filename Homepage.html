package part2; import org.apache.hadoop.io.Text; import org.apache.hadoop.mapreduce.Reducer; import java.io.IOException; import java.util.HashSet; public class IDFReducer extends Reducer { private Text result = new Text(); @Override protected void reduce(Text key, Iterable values, Context context) throws IOException, InterruptedException { // Initialize HashSet to track unique document IDs HashSet uniqueDocs = new HashSet<>(); // Loop through the values and add document IDs to the set (no duplicates) for (Text val : values) { uniqueDocs.add(val.toString()); // Add docId to the set } int docCount = uniqueDocs.size(); // Document frequency (DF) is the size of the set // Ensure DF is not zero to avoid division by zero issues if (docCount == 0) { docCount = 1; // If no documents contain the term, we can assign DF=1 to avoid division by zero } // Calculate IDF: IDF = log(N / DF), where N is the total number of documents (assumed 10) double idf = Math.log(10.0 / docCount); // Set the IDF result result.set(String.valueOf(idf)); // Emit the term and its IDF value context.write(key, result); } }
