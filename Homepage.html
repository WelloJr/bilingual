package part2; import org.apache.hadoop.io.Text; import org.apache.hadoop.mapreduce.Reducer; import java.io.IOException; public class IDFReducer extends Reducer { private Text result = new Text(); @Override protected void reduce(Text key, Iterable values, Context context) throws IOException, InterruptedException { int docCount = 0; // Use a HashSet to count unique documents HashSet uniqueDocs = new HashSet<>(); for (Text val : values) { uniqueDocs.add(val.toString()); // Each docId should be counted only once } docCount = uniqueDocs.size(); // The DF is the size of the uniqueDocs set // Calculate IDF: IDF = log(N / DF) where N is the total number of documents (10) double idf = Math.log(10.0 / (docCount + 1)); // Avoid division by zero by adding 1 to DF result.set(String.valueOf(idf)); // Set the IDF value context.write(key, result); // Emit the term and its IDF value } }
