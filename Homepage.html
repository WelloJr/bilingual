package part2; import org.apache.hadoop.io.Text; import org.apache.hadoop.mapreduce.Reducer; import java.io.IOException; import java.util.HashMap; import java.util.Map; public class TFReducer extends Reducer { private Text result = new Text(); @Override protected void reduce(Text key, Iterable values, Context context) throws IOException, InterruptedException { Map docCountMap = new HashMap<>(); // Loop through all the values (docId:positions) for (Text val : values) { String[] docAndPosition = val.toString().split(":"); String docId = docAndPosition[0]; // Extract the document ID // Check if the document ID already exists in the map, and update its count if (docCountMap.containsKey(docId)) { docCountMap.put(docId, docCountMap.get(docId) + 1); } else { docCountMap.put(docId, 1); // Initialize the count for the document } } // Build the result string StringBuilder output = new StringBuilder(); for (Map.Entry entry : docCountMap.entrySet()) { output.append(entry.getKey()).append(": ").append(entry.getValue()).append(" ; "); } // Write the output in the desired format result.set(output.toString().trim()); context.write(key, result); } }
