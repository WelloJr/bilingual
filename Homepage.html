package part2; import org.apache.hadoop.io.Text; import org.apache.hadoop.mapreduce.Mapper; import java.io.IOException; import java.util.HashSet; public class IDFMapper extends Mapper { private Text term = new Text(); private Text docId = new Text(); @Override protected void map(Object key, Text value, Context context) throws IOException, InterruptedException { String line = value.toString().trim(); String[] parts = line.split("\\s+", 2); // Split the term and doc data String termText = parts[0]; // The term String docData = parts[1]; // The document data (docId: positions) // Use a HashSet to ensure each document is counted only once HashSet uniqueDocs = new HashSet<>(); String[] docs = docData.split(";"); for (String doc : docs) { String docIdPart = doc.split(":")[0]; // Extract the docId uniqueDocs.add(docIdPart); // Add the document to the set (to avoid duplicates) } // Emit term and document id (we only emit each document once per term) for (String doc : uniqueDocs) { term.set(termText); // Set the term docId.set(doc); // Set the document ID context.write(term, docId); // Emit the pair (term, docId) } } }
