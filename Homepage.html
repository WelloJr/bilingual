package part2; import org.apache.hadoop.io.Text; import org.apache.hadoop.mapreduce.Reducer; import java.io.IOException; import java.util.HashSet; public class IDFReducer extends Reducer { private Text result = new Text(); @Override protected void reduce(Text key, Iterable values, Context context) throws IOException, InterruptedException { // Use a Set to store unique document IDs HashSet uniqueDocs = new HashSet<>(); // Iterate through the values (docId) and add them to the Set for (Text value : values) { String docId = value.toString().trim(); // Document ID from the Mapper output uniqueDocs.add(docId); // Only unique docIds will be added } // Calculate IDF: log(Total Docs / Docs Containing Term) int docCount = uniqueDocs.size(); // Number of documents containing the term double idf = Math.log10(10.0 / docCount); // Assuming there are 10 documents in total // Write the term and its IDF value result.set(String.valueOf(idf)); context.write(key, result); } }
