package part2; import org.apache.hadoop.io.Text; import org.apache.hadoop.mapreduce.Reducer; import java.io.IOException; import java.util.HashSet; public class IDFReducer extends Reducer { private Text result = new Text(); @Override protected void reduce(Text key, Iterable values, Context context) throws IOException, InterruptedException { // Use a HashSet to track unique document IDs HashSet uniqueDocs = new HashSet<>(); // Count unique document IDs for the term for (Text val : values) { uniqueDocs.add(val.toString()); // Add docId to the set } int docCount = uniqueDocs.size(); // Document frequency (DF) is the size of the uniqueDocs set // Check DF to avoid division by zero or an incorrect log value if (docCount == 0) { docCount = 1; // Avoid division by zero, though ideally DF should never be zero } // Calculate IDF: IDF = log(N / DF), where N is the total number of documents (assumed 10) double idf = Math.log(10.0 / docCount); result.set(String.valueOf(idf)); // Set the IDF value context.write(key, result); // Emit the term and its IDF value } }
