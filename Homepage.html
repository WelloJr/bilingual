package part2; import org.apache.hadoop.io.Text; import org.apache.hadoop.mapreduce.Mapper; import java.io.IOException; import java.util.StringTokenizer; public class TFMapper extends Mapper { private Text term = new Text(); private Text docAndCount = new Text(); @Override protected void map(Object key, Text value, Context context) throws IOException, InterruptedException { // Input format: angels < angels doc7: 1 ; doc9: 1 ; doc8: 1 ; > String line = value.toString().trim(); // Remove the surrounding < > from the input data if (line.startsWith("<") && line.endsWith(">")) { line = line.substring(1, line.length() - 1).trim(); } // Split the line into term and doc data String[] parts = line.split(" ", 2); String termText = parts[0]; // The term (e.g., angels) String docData = parts[1].split(";")[0].trim(); // Document data without the semicolon // Tokenize the document data to get docId and positions StringTokenizer tokenizer = new StringTokenizer(docData); while (tokenizer.hasMoreTokens()) { String token = tokenizer.nextToken(); String[] docAndPos = token.split(":"); if (docAndPos.length == 2) { String docId = docAndPos[0]; // Document ID (e.g., doc7) String positions = docAndPos[1]; // Positions (e.g., 1, 4, 7) // Emit the term as key and docId:positions as value term.set(termText); docAndCount.set(docId + ":" + positions); context.write(term, docAndCount); // Emit the key-value pair } } } }
